ALTER TABLE PO.USER_ACTIVITY
 DROP PRIMARY KEY CASCADE;

DROP TABLE PO.USER_ACTIVITY CASCADE CONSTRAINTS;

CREATE TABLE PO.USER_ACTIVITY
(
  USERNAME       VARCHAR2(100 BYTE)             NOT NULL,
  IP_ADDRESS     VARCHAR2(50 BYTE)              NOT NULL,
  ACTIVITY_DATE  DATE                           DEFAULT NULL                  NOT NULL,
  STATUS_ERROR   NUMBER(1)                      DEFAULT 0                     NOT NULL,
  KETERANGAN     VARCHAR2(500 BYTE)
)
TABLESPACE PONLINE
PCTUSED    0
PCTFREE    10
INITRANS   1
MAXTRANS   255
STORAGE    (
            INITIAL          64K
            NEXT             1M
            MINEXTENTS       1
            MAXEXTENTS       UNLIMITED
            PCTINCREASE      0
            BUFFER_POOL      DEFAULT
           )
LOGGING 
NOCOMPRESS 
NOCACHE
NOPARALLEL
MONITORING;

COMMENT ON COLUMN PO.USER_ACTIVITY.STATUS_ERROR IS '0=NON_ACTIVE,1=ACTIVE';
