DROP TABLE PO.USER_SETTING_COLUMN CASCADE CONSTRAINTS;

CREATE TABLE PO.USER_SETTING_COLUMN
(
  USERNAME     VARCHAR2(100 BYTE)               NOT NULL,
  NOP          VARCHAR2(100 BYTE)               NOT NULL,
  COLUMN_NAME  VARCHAR2(50 CHAR),
  COLUMN_TEXT  VARCHAR2(50 CHAR)
)
TABLESPACE PONLINE
PCTUSED    0
PCTFREE    10
INITRANS   1
MAXTRANS   255
STORAGE    (
            INITIAL          64K
            NEXT             1M
            MINEXTENTS       1
            MAXEXTENTS       UNLIMITED
            PCTINCREASE      0
            BUFFER_POOL      DEFAULT
           )
LOGGING 
NOCOMPRESS 
NOCACHE
NOPARALLEL
MONITORING;


DROP PUBLIC SYNONYM USER_SETTING_COLUMN;

CREATE PUBLIC SYNONYM USER_SETTING_COLUMN FOR PO.USER_SETTING_COLUMN;


GRANT SELECT ON PO.USER_SETTING_COLUMN TO YUDA;
